001package software.leonov.util.gson;
002
003import static com.google.common.base.Preconditions.checkNotNull;
004
005import java.io.IOException;
006import java.lang.reflect.Type;
007import java.util.Optional;
008
009import com.google.gson.Gson;
010import com.google.gson.TypeAdapter;
011import com.google.gson.reflect.TypeToken;
012import com.google.gson.stream.JsonReader;
013import com.google.gson.stream.JsonToken;
014import com.google.gson.stream.JsonWriter;
015
016/**
017 * A {@link TypeAdapter} which can serialize and deserialize of {@link Optional} values to and from JSON.
018 * <p>
019 * {@link Optional#empty() Empty} {@code Optional}s are serialized as JSON-encoded {@link JsonToken#NULL NULL}s while
020 * {@link Optional#isPresent() present} values are serialized using an {@link Gson#getAdapter(TypeToken) adapter
021 * obtained} from the provided {@code Gson} instance.
022 *
023 * @param <E> the type of the element contained within the {@code Optional}
024 */
025public final class OptionalTypeAdapter<E> extends TypeAdapter<Optional<E>> {
026
027    private final TypeAdapter<E> adapter;
028
029    /**
030     * Constructs a new {@code OptionalTypeAdapter}.
031     * <p>
032     * While this constructor can be invoked manually, it will most commonly be invoked by
033     * {@link OptionalTypeAdapterFactory}, which in turn will be registered with the provided {@link Gson} instance.
034     *
035     * @param gson a {@code Gson} instance which can handle the serialization and deserialization of the type of element
036     *             contained within the {@code Optional}
037     * @param type the {@link Type} of the element contained in the {@code Optional}
038     */
039    @SuppressWarnings("unchecked")
040    public OptionalTypeAdapter(final Gson gson, final Type type) {
041        checkNotNull(gson, "gson == null");
042        checkNotNull(type, "type == null");
043
044        this.adapter = (TypeAdapter<E>) gson.getAdapter(TypeToken.get(type));
045    }
046
047//    /**
048//     * Writes the specified {@code Optional} value to the {@code JsonWriter}.
049//     * <p>
050//     * <b>Warning:</b> Unlike the {@code write} method in most adapters which accepts {@code null} values an
051//     * {@code Optional} value cannot itself be {@code null}.
052//     * 
053//     * @param out   the specified {@code JsonWriter}
054//     * @param value the {@code Optional} value to write
055//     * @throws NullPointerException if the {@code Optional} value itself is {@code null}
056//     */
057    @Override
058    public void write(final JsonWriter out, final Optional<E> value) throws IOException {
059        checkNotNull(out, "out == null");
060        checkNotNull(value, "optional == null"); // an Optional itself should never be null
061
062        if (value.isPresent())
063            adapter.write(out, value.get());
064        else
065            out.nullValue();
066    }
067
068    @Override
069    public Optional<E> read(final JsonReader in) throws IOException {
070        checkNotNull(in, "in == null");
071
072        if (in.peek() == JsonToken.NULL) {
073            in.nextNull();
074            return Optional.empty();
075        }
076        return Optional.ofNullable(adapter.read(in));
077    }
078
079}
