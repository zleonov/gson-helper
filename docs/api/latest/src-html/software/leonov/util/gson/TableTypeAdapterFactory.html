001package software.leonov.util.gson;
002
003import static com.google.common.base.Preconditions.checkNotNull;
004
005import java.lang.reflect.ParameterizedType;
006import java.lang.reflect.Type;
007
008import com.google.common.collect.HashBasedTable;
009import com.google.common.collect.RowSortedTable;
010import com.google.common.collect.Table;
011import com.google.common.collect.TreeBasedTable;
012import com.google.gson.Gson;
013import com.google.gson.TypeAdapter;
014import com.google.gson.TypeAdapterFactory;
015import com.google.gson.reflect.TypeToken;
016
017/**
018 * Creates {@link TypeAdapter}s for serializing and deserializing {@link Table} values to and from JSON.
019 * <p>
020 * <strong>Supported {@code Table} Types:</strong>
021 * <ul>
022 * <li>{@link HashBasedTable} (default for deserialization)</li>
023 * <li>{@link TreeBasedTable}</li>
024 * </ul>
025 * <p>
026 * <b>Serialization/Deserialization Behavior:</b>
027 * <p>
028 * All supported {@code Table} implementations can be serialized to JSON. Attempting to deserialize unsupported
029 * {@code Table} implementations will result in an {@link IllegalArgumentException}.
030 *
031 * @author Zhenya Leonov
032 */
033public final class TableTypeAdapterFactory implements TypeAdapterFactory {
034
035    @SuppressWarnings("unchecked")
036    @Override
037    public <T> TypeAdapter<T> create(final Gson gson, final TypeToken<T> typeToken) {
038        checkNotNull(gson, "gson == null");
039        checkNotNull(typeToken, "typeToken == null");
040
041        final Class<? super T> clazz = typeToken.getRawType();
042
043        if (Table.class.isAssignableFrom(clazz)) {
044            final Type type = typeToken.getType();
045
046            final Type rowKeyType;
047            final Type columnKeyType;
048            final Type valueType;
049
050            if (type instanceof ParameterizedType) {
051                final ParameterizedType paramType = (ParameterizedType) type;
052                rowKeyType    = paramType.getActualTypeArguments()[0];
053                columnKeyType = paramType.getActualTypeArguments()[1];
054                valueType     = paramType.getActualTypeArguments()[2];
055            } else {
056                rowKeyType    = Object.class;
057                columnKeyType = Object.class;
058                valueType     = Object.class;
059            }
060
061            if (clazz.equals(Table.class) || clazz.equals(HashBasedTable.class))
062                return (TypeAdapter<T>) new TableTypeAdapter<>(gson, rowKeyType, columnKeyType, valueType, () -> HashBasedTable.create());
063            else if (clazz.equals(RowSortedTable.class) || clazz.equals(TreeBasedTable.class))
064                return (TypeAdapter<T>) new TableTypeAdapter<>(gson, rowKeyType, columnKeyType, valueType, () -> TreeBasedTable.create());
065
066            throw new IllegalArgumentException(String.format("%s is not supported; try one of [%s, %s]", clazz.getSimpleName(), HashBasedTable.class.getSimpleName(), TreeBasedTable.class.getSimpleName()));
067        }
068
069        return null;
070    }
071}
