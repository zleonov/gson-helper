001package software.leonov.util.gson.factory;
002
003import static com.google.common.base.Preconditions.checkNotNull;
004
005import java.lang.reflect.ParameterizedType;
006import java.lang.reflect.Type;
007import java.util.Optional;
008
009import com.google.gson.Gson;
010import com.google.gson.TypeAdapter;
011import com.google.gson.TypeAdapterFactory;
012import com.google.gson.reflect.TypeToken;
013
014import software.leonov.util.gson.adapter.OptionalTypeAdapter;
015
016/**
017 * Creates {@link TypeAdapter}s which can handle the serialization and deserialization of {@link Optional} values to and
018 * from JSON.
019 * 
020 * @author Zhenya Leonov
021 */
022public final class OptionalTypeAdapterFactory implements TypeAdapterFactory {
023
024    @SuppressWarnings("unchecked")
025    @Override
026    public <T> TypeAdapter<T> create(final Gson gson, final TypeToken<T> typeToken) {
027        checkNotNull(gson, "gson == null");
028        checkNotNull(typeToken, "typeToken == null");
029
030        final Class<? super T> clazz = typeToken.getRawType();
031
032        if (Optional.class.isAssignableFrom(clazz)) {
033            final Type type = typeToken.getType();
034
035            final Type elementType = type instanceof ParameterizedType ? ((ParameterizedType) type).getActualTypeArguments()[0] : Object.class;
036
037            return (TypeAdapter<T>) new OptionalTypeAdapter<>(gson, elementType);
038        }
039
040        return null;
041    }
042
043}
