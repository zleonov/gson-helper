001package software.leonov.util.gson;
002
003import static com.google.common.base.Preconditions.checkNotNull;
004
005import java.lang.reflect.ParameterizedType;
006import java.lang.reflect.Type;
007
008import com.google.common.collect.ConcurrentHashMultiset;
009import com.google.common.collect.HashMultiset;
010import com.google.common.collect.LinkedHashMultiset;
011import com.google.common.collect.Multiset;
012import com.google.common.collect.SortedMultiset;
013import com.google.common.collect.TreeMultiset;
014import com.google.gson.Gson;
015import com.google.gson.TypeAdapter;
016import com.google.gson.TypeAdapterFactory;
017import com.google.gson.reflect.TypeToken;
018
019/**
020 * Creates {@link TypeAdapter}s for serializing and deserializing {@link Multiset} values to and from JSON.
021 * <p>
022 * <strong>Supported {@code Multiset} Types:</strong>
023 * <ul>
024 * <li>{@link LinkedHashMultiset} (default for deserialization)</li>
025 * <li>{@link HashMultiset}</li>
026 * <li>{@link TreeMultiset}</li>
027 * <li>{@link ConcurrentHashMultiset}</li>
028 * </ul>
029 * <p>
030 * <b>Serialization/Deserialization Behavior:</b>
031 * <p>
032 * All supported {@code Multiset} implementations can be serialized to JSON. Attempting to deserialize unsupported
033 * {@code Multiset} implementations will result in an {@link IllegalArgumentException}.
034 *
035 * @author Zhenya Leonov
036 */
037public final class MultisetTypeAdapterFactory implements TypeAdapterFactory {
038
039    @SuppressWarnings("unchecked")
040    @Override
041    public <T> TypeAdapter<T> create(final Gson gson, final TypeToken<T> typeToken) {
042        checkNotNull(gson, "gson == null");
043        checkNotNull(typeToken, "typeToken == null");
044
045        final Class<? super T> clazz = typeToken.getRawType();
046
047        if (Multiset.class.isAssignableFrom(clazz)) {
048            final Type type = typeToken.getType();
049
050            final Type elementType = type instanceof ParameterizedType ? ((ParameterizedType) type).getActualTypeArguments()[0] : Object.class;
051
052            if (clazz.equals(HashMultiset.class))
053                return (TypeAdapter<T>) new MultisetTypeAdapter<>(gson, elementType, () -> HashMultiset.create());
054            else if (clazz.equals(Multiset.class) || clazz.equals(LinkedHashMultiset.class))
055                return (TypeAdapter<T>) new MultisetTypeAdapter<>(gson, elementType, () -> LinkedHashMultiset.create());
056            else if (clazz.equals(SortedMultiset.class) || clazz.equals(TreeMultiset.class))
057                return (TypeAdapter<T>) new MultisetTypeAdapter<>(gson, elementType, () -> TreeMultiset.create());
058            else if (clazz.equals(ConcurrentHashMultiset.class))
059                return (TypeAdapter<T>) new MultisetTypeAdapter<>(gson, elementType, () -> ConcurrentHashMultiset.create());
060
061            throw new IllegalArgumentException(String.format("%s is not supported; try one of [%s, %s, %s, %s]", clazz.getSimpleName(), HashMultiset.class.getSimpleName(), LinkedHashMultiset.class.getSimpleName(),
062                    TreeMultiset.class.getSimpleName(), ConcurrentHashMultiset.class.getSimpleName()));
063        }
064        return null;
065    }
066}
