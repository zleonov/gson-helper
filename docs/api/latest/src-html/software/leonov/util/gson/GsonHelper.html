033/**
034 * A collection of static utility methods that simplify common serialization and deserialization operations using
035 * <a href="https://github.com/google/gson">Google Gson</a>.
036 * <p>
037 * At the core of {@code GsonHelper} is a {@link #getGson() default} {@code Gson} instance (suitable for general use
038 * cases) which supports Java 8+
039 * <a href="https://docs.oracle.com/javase/8/docs/api/java/time/package-summary.html">java.time</a> classes,
040 * {@link Optional}s, Guava's {@link Multiset}, {@link Multimap}, and {@link Table} collection types,
041 * {@link GsonBuilder#serializeNulls() serializes nulls}, {@link GsonBuilder#disableHtmlEscaping() does not escape HTML
042 * characters}, and outputs {@link GsonBuilder#setPrettyPrinting() <i>pretty</i> printed} JSON text.
043 * <p>
044 * If you would like to use your own {@code Gson} instance you can register adapters for
045 * {@link #registerAll(GsonBuilder) Guava's collection types} along with {@link Converters#registerAll(GsonBuilder)
046 * gson-javatime-serialisers} manually.
047 * <p>
048 * For example: <pre><code class="line-numbers match-braces language-java">
049 *   private static final Gson GSON = GsonHelper.registerAll(Converters.registerAll(new GsonBuilder())
050 *           .serializeNulls()
051 *           .setPrettyPrinting()
052 *           .disableHtmlEscaping())
053 *           .enableComplexMapKeySerialization()
054 *           .create();
055 * </code></pre>
056 * 
057 * @author Zhenya Leonov
058 */
059public final class GsonHelper {

098     * Registers {@link OptionalTypeAdapterFactory}, {@link MultisetTypeAdapterFactory}, {@link MultimapTypeAdapterFactory},
099     * and {@link TableTypeAdapterFactory} with the specified {@link GsonBuilder}.
100     * 
101     * @param builder the specified GSON builder
102     * @return the specified GSON builder
103     */
104    public static GsonBuilder registerAll(final GsonBuilder builder) {
105        checkNotNull(builder, "builder == null");
106
107        // @formatter:off
108        builder.registerTypeAdapterFactory(new OptionalTypeAdapterFactory())
109               .registerTypeAdapterFactory(new MultisetTypeAdapterFactory())
110               .registerTypeAdapterFactory(new MultimapTypeAdapterFactory())
111               .registerTypeAdapterFactory(new TableTypeAdapterFactory());
112        // @formatter:on
113
114        return builder;
115    }

424    /**
425     * Copies all {@code JsonToken}s from the specified {@code JsonReader} to the given {@code JsonWriter}.
426     * <p>
427     * Does not close the reader or the writer.
428     * 
429     * @param <W>  the type of {@code JsonWriter}
430     * @param from the specified {@code JsonReader}
431     * @param to   the given {@code JsonWriter}
432     * @return the given {@code JsonWriter}
433     * @throws IOException if an I/O error occurs
434     */
435    public static <W extends JsonWriter> W copy(final JsonReader from, final W to) throws IOException {
436        checkNotNull(from, "from == null");
437        checkNotNull(to, "to == null");
438
439        for (JsonToken peek = from.peek(); peek != JsonToken.END_DOCUMENT; peek = from.peek()) {
440            if (peek == JsonToken.BEGIN_ARRAY) {
441                from.beginArray();
442                to.beginArray();
443            } else if (peek == JsonToken.BEGIN_OBJECT) {
444                from.beginObject();
445                to.beginObject();
446            } else if (peek == JsonToken.BOOLEAN)
447                to.value(from.nextBoolean());
448            else if (peek == JsonToken.END_ARRAY) {
449                from.endArray();
450                to.endArray();
451            } else if (peek == JsonToken.END_OBJECT) {
452                from.endObject();
453                to.endObject();
454            } else if (peek == JsonToken.NAME)
455                to.name(from.nextName());
456            else if (peek == JsonToken.NULL) {
457                from.nextNull();
458                to.nullValue();
459            } else if (peek == JsonToken.NUMBER)
460                to.value(new BigDecimal(from.nextString()));
461            else if (peek == JsonToken.STRING)
462                to.value(from.nextString());
463        }
464
465        to.flush();
466        return to;
467    }

483     * @param out     the given {@code JsonWriter}

488        checkNotNull(element, "element == null");
489        checkNotNull(out, "out == null");
490        return copy(new JsonTreeReader(element), out);
