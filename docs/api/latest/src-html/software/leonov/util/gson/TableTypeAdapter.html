001package software.leonov.util.gson;
002
003import static com.google.common.base.Preconditions.checkNotNull;
004
005import java.io.IOException;
006import java.lang.reflect.Type;
007import java.util.Map;
008import java.util.function.Supplier;
009
010import com.google.common.collect.HashBasedTable;
011import com.google.common.collect.Table;
012import com.google.gson.Gson;
013import com.google.gson.GsonBuilder;
014import com.google.gson.TypeAdapter;
015import com.google.gson.reflect.TypeToken;
016import com.google.gson.stream.JsonReader;
017import com.google.gson.stream.JsonToken;
018import com.google.gson.stream.JsonWriter;
019
020/**
021 * A {@link TypeAdapter} which can serialize and deserialize of {@link Table} values to and from JSON.
022 * <p>
023 * This adapter supports complex key serialization if the provided {@code Gson} instance
024 * {@link GsonBuilder#enableComplexMapKeySerialization() supports} it.
025 *
026 * @param <R> the type of row keys
027 * @param <C> the type of column keys
028 * @param <V> the type of mapped values
029 */
030public final class TableTypeAdapter<R, C, V> extends TypeAdapter<Table<R, C, V>> {
031
032    private final Type mapType = new TypeToken<Map<R, Map<C, V>>>() {
033    }.getType();
034
035    private final Gson                               gson;
036    private final TypeAdapter<R>                     rowKeyAdapter;
037    private final TypeAdapter<C>                     columnKeyAdapter;
038    private final TypeAdapter<V>                     valueAdapter;
039    private final Supplier<? extends Table<R, C, V>> supplier;
040
041    /**
042     * Constructs a new {@code TableTypeAdapter} which can serialize and deserialize {@code Table} values.
043     * 
044     * @param gson       a {@code Gson} instance which can handle the serialization and deserialization of the type of keys
045     *                   and values contained within the {@code Table}
046     * @param rowType    the {@link Type} of row keys
047     * @param columnType the {@link Type} of column keys
048     * @param valueType  the {@link Type} of mapped values
049     */
050    public TableTypeAdapter(final Gson gson, final Type rowType, final Type columnType, final Type valueType) {
051        this(gson, rowType, columnType, valueType, () -> HashBasedTable.create());
052    }
053
054    /**
055     * Constructs a new {@code TableTypeAdapter} which can serialize and deserialize {@code Table} values.
056     * <p>
057     * While this constructor can be invoked manually, it will most commonly be invoked by {@link TableTypeAdapterFactory},
058     * which in turn will be registered with the provided {@link Gson} instance.
059     * 
060     * @param gson       a {@code Gson} instance which can handle the serialization and deserialization of the type of keys
061     *                   and values contained within the {@code Table}
062     * @param rowType    the {@link Type} of row keys
063     * @param columnType the {@link Type} of column keys
064     * @param valueType  the {@link Type} of mapped values
065     * @param supplier   a {@link Supplier} which returns empty {@code Table} instances
066     */
067    @SuppressWarnings("unchecked")
068    public TableTypeAdapter(final Gson gson, final Type rowType, final Type columnType, final Type valueType, final Supplier<? extends Table<R, C, V>> supplier) {
069        checkNotNull(gson, "gson == null");
070        checkNotNull(rowType, "rowType == null");
071        checkNotNull(columnType, "columnType == null");
072        checkNotNull(valueType, "valueType == null");
073        checkNotNull(supplier, "supplier == null");
074
075        this.gson             = gson;
076        this.rowKeyAdapter    = (TypeAdapter<R>) gson.getAdapter(TypeToken.get(rowType));
077        this.columnKeyAdapter = (TypeAdapter<C>) gson.getAdapter(TypeToken.get(columnType));
078        this.valueAdapter     = (TypeAdapter<V>) gson.getAdapter(TypeToken.get(valueType));
079        this.supplier         = supplier;
080    }
081
082    @Override
083    public void write(final JsonWriter out, final Table<R, C, V> table) throws IOException {
084        checkNotNull(out, "out == null");
085
086        if (table == null) {
087            out.nullValue();
088            return;
089        }
090
091        gson.toJson(table.rowMap(), mapType, out);
092    }
093
094    @Override
095    public Table<R, C, V> read(final JsonReader in) throws IOException {
096        checkNotNull(in, "in == null");
097
098        final Table<R, C, V> table = supplier.get();
099
100        if (in.peek() == JsonToken.NULL) {
101            in.nextNull();
102            return null;
103        } else if (in.peek() == JsonToken.BEGIN_ARRAY) { // see GsonBuilder.enableComplexMapKeySerialization()
104            in.beginArray(); // elements
105            while (in.hasNext()) {
106                in.beginArray(); // element
107                final R rowKey = rowKeyAdapter.read(in); // row key
108
109                if (in.peek() == JsonToken.BEGIN_ARRAY) { // see GsonBuilder.enableComplexMapKeySerialization()
110                    in.beginArray(); // elements
111                    while (in.hasNext()) {
112                        in.beginArray(); // element
113                        final C colKey = columnKeyAdapter.read(in); // key
114                        final V value  = valueAdapter.read(in);     // value
115                        in.endArray();
116                        table.put(rowKey, colKey, value);
117                    }
118                    in.endArray();
119                } else {
120                    in.beginObject();
121                    final String colKeyJson = "\"" + in.nextName() + "\"";          // row key
122                    final C      colKey     = columnKeyAdapter.fromJson(colKeyJson);
123                    final V      value      = valueAdapter.read(in);
124                    in.endObject();
125                    table.put(rowKey, colKey, value);
126                }
127
128                in.endArray();
129            }
130            in.endArray();
131        } else {
132            in.beginObject(); // row map
133            while (in.hasNext()) {
134
135                final String rowKeyJson = "\"" + in.nextName() + "\"";       // row key
136                final R      rowKey     = rowKeyAdapter.fromJson(rowKeyJson);
137
138                in.beginObject(); // column map
139                while (in.hasNext()) {
140                    final String columnKeyJson = "\"" + in.nextName() + "\"";             // column key
141                    final C      columnKey     = columnKeyAdapter.fromJson(columnKeyJson);
142
143                    final V value = valueAdapter.read(in); // value
144
145                    table.put(rowKey, columnKey, value);
146                }
147                in.endObject(); // end of column map
148            }
149            in.endObject(); // end of row map
150        }
151        return table;
152    }
153}
