001package software.leonov.util.gson.factory;
002
003import static com.google.common.base.Preconditions.checkNotNull;
004
005import java.lang.reflect.ParameterizedType;
006import java.lang.reflect.Type;
007
008import com.google.common.collect.ArrayListMultimap;
009import com.google.common.collect.HashMultimap;
010import com.google.common.collect.LinkedHashMultimap;
011import com.google.common.collect.LinkedListMultimap;
012import com.google.common.collect.ListMultimap;
013import com.google.common.collect.Multimap;
014import com.google.common.collect.SetMultimap;
015import com.google.common.collect.SortedSetMultimap;
016import com.google.common.collect.TreeMultimap;
017import com.google.gson.Gson;
018import com.google.gson.TypeAdapter;
019import com.google.gson.TypeAdapterFactory;
020import com.google.gson.reflect.TypeToken;
021
022import software.leonov.util.gson.adapter.MultimapTypeAdapter;
023
024/**
025 * Creates {@link TypeAdapter}s for serializing and deserializing {@link Multimap} values to and from JSON.
026 * <p>
027 * <strong>Supported {@code Multiset} Types:</strong>
028 * <ul>
029 * <li>{@link ArrayListMultimap} (default for deserialization)</li>
030 * <li>{@link HashMultimap}</li>
031 * <li>{@link LinkedHashMultimap}</li>
032 * <li>{@link TreeMultimap}</li>
033 * <li>{@link LinkedListMultimap}</li>
034 * </ul>
035 * <p>
036 * <b>Serialization/Deserialization Behavior:</b>
037 * <p>
038 * All supported {@code Multimap} implementations can be serialized to JSON. Attempting to deserialize unsupported
039 * {@code Multimap} implementations will result in an {@link IllegalArgumentException}.
040 *
041 * @author Zhenya Leonov
042 */
043public final class MultimapTypeAdapterFactory implements TypeAdapterFactory {
044
045    @SuppressWarnings("unchecked")
046    @Override
047    public <T> TypeAdapter<T> create(final Gson gson, final TypeToken<T> typeToken) {
048        checkNotNull(gson, "gson == null");
049        checkNotNull(typeToken, "typeToken == null");
050
051        final Class<? super T> clazz = typeToken.getRawType();
052
053        if (Multimap.class.isAssignableFrom(clazz)) {
054            final Type type = typeToken.getType();
055
056            final Type keyType;
057            final Type valueType;
058
059            if (type instanceof ParameterizedType) {
060                final ParameterizedType paramType = (ParameterizedType) type;
061                keyType   = paramType.getActualTypeArguments()[0];
062                valueType = paramType.getActualTypeArguments()[1];
063            } else {
064                keyType   = Object.class;
065                valueType = Object.class;
066            }
067
068            if (clazz.equals(Multimap.class) || clazz.equals(ListMultimap.class) || clazz.equals(ArrayListMultimap.class))
069                return (TypeAdapter<T>) new MultimapTypeAdapter<>(gson, keyType, valueType, () -> ArrayListMultimap.create());
070            else if (clazz.equals(SetMultimap.class) || clazz.equals(HashMultimap.class))
071                return (TypeAdapter<T>) new MultimapTypeAdapter<>(gson, keyType, valueType, () -> HashMultimap.create());
072            else if (clazz.equals(LinkedHashMultimap.class))
073                return (TypeAdapter<T>) new MultimapTypeAdapter<>(gson, keyType, valueType, () -> LinkedHashMultimap.create());
074            else if (clazz.equals(SortedSetMultimap.class) || clazz.equals(TreeMultimap.class))
075                return (TypeAdapter<T>) new MultimapTypeAdapter<>(gson, keyType, valueType, () -> TreeMultimap.create());
076            else if (clazz.equals(LinkedListMultimap.class))
077                return (TypeAdapter<T>) new MultimapTypeAdapter<>(gson, keyType, valueType, () -> LinkedListMultimap.create());
078
079            throw new IllegalArgumentException(String.format("%s is not supported; try one of [%s, %s, %s, %s, %s]", clazz.getSimpleName(), ArrayListMultimap.class.getSimpleName(), HashMultimap.class.getSimpleName(),
080                    LinkedHashMultimap.class.getSimpleName(), TreeMultimap.class.getSimpleName(), LinkedListMultimap.class.getSimpleName()));
081        }
082
083        return null;
084    }
085}
