001package software.leonov.util.gson;
002
003import static com.google.common.base.Preconditions.checkNotNull;
004
005import java.io.IOException;
006import java.lang.reflect.Type;
007import java.util.Collection;
008import java.util.Map;
009import java.util.function.Supplier;
010
011import com.google.common.collect.ArrayListMultimap;
012import com.google.common.collect.Multimap;
013import com.google.gson.Gson;
014import com.google.gson.GsonBuilder;
015import com.google.gson.TypeAdapter;
016import com.google.gson.reflect.TypeToken;
017import com.google.gson.stream.JsonReader;
018import com.google.gson.stream.JsonToken;
019import com.google.gson.stream.JsonWriter;
020
021/**
022 * A {@link TypeAdapter} which can serialize and deserialize of {@link Multimap} values to and from JSON.
023 * <p>
024 * This adapter supports complex key serialization if the provided {@code Gson} instance
025 * {@link GsonBuilder#enableComplexMapKeySerialization() supports} it.
026 *
027 * @param <K> the type of keys
028 * @param <V> the type of mapped values
029 */
030public final class MultimapTypeAdapter<K, V> extends TypeAdapter<Multimap<K, V>> {
031
032    private final Type mapType = new TypeToken<Map<K, Collection<V>>>() {
033    }.getType();
034
035    private final Gson                               gson;
036    private final TypeAdapter<K>                     keyAdapter;
037    private final TypeAdapter<V>                     valueAdapter;
038    private final Supplier<? extends Multimap<K, V>> supplier;
039
040    /**
041     * Constructs a new {@code MultimapTypeAdapter} which can serialize and deserialize {@code Multimap} values as
042     * {@link ArrayListMultimap}s.
043     *
044     * @param gson      a {@code Gson} instance which can handle the serialization and deserialization of the type of keys
045     *                  and values contained within the {@code Multimap}
046     * @param keyType   the {@link Type} of keys
047     * @param valueType the {@link Type} of mapped values
048     */
049    public MultimapTypeAdapter(final Gson gson, final Type keyType, final Type valueType) {
050        this(gson, keyType, valueType, () -> ArrayListMultimap.create());
051    }
052
053    /**
054     * Constructs a new {@code MultimapTypeAdapter} which can serialize and deserialize {@code Multimap} values.
055     * <p>
056     * While this constructor can be invoked manually, it will most commonly be invoked by
057     * {@link MultimapTypeAdapterFactory}, which in turn will be registered with the provided {@link Gson} instance.
058     *
059     * @param gson      a {@code Gson} instance which can handle the serialization and deserialization of the type of keys
060     *                  and values contained within the {@code Multimap}
061     * @param keyType   the {@link Type} of keys
062     * @param valueType the {@link Type} of mapped values
063     * @param supplier  a {@link Supplier} which returns empty {@code Multimap} instances
064     */
065    @SuppressWarnings("unchecked")
066    public MultimapTypeAdapter(final Gson gson, final Type keyType, final Type valueType, final Supplier<? extends Multimap<K, V>> supplier) {
067        checkNotNull(gson, "gson == null");
068        checkNotNull(keyType, "keyType == null");
069        checkNotNull(valueType, "valueType == null");
070        checkNotNull(supplier, "supplier == null");
071
072        this.gson         = gson;
073        this.keyAdapter   = (TypeAdapter<K>) gson.getAdapter(TypeToken.get(keyType));
074        this.valueAdapter = (TypeAdapter<V>) gson.getAdapter(TypeToken.get(valueType));
075        this.supplier     = supplier;
076    }
077
078    /**
079     * Writes the specified {@code Multimap} value to the {@code JsonWriter}.
080     * 
081     * @param out   the specified {@code JsonWriter}
082     * @param value the {@code Multimap} value to write
083     */
084    @Override
085    public void write(final JsonWriter out, final Multimap<K, V> value) throws IOException {
086        checkNotNull(out, "out == null");
087
088        if (value == null) {
089            out.nullValue();
090            return;
091        }
092
093        gson.toJson(value.asMap(), mapType, out);
094    }
095
096    /**
097     * Returns the next {@code Multimap} value read from the specified {@code JsonReader}.
098     * 
099     * @return the next {@code Multimap} value read from the specified {@code JsonReader}
100     */
101    @Override
102    public Multimap<K, V> read(final JsonReader in) throws IOException {
103        checkNotNull(in, "in == null");
104
105        final Multimap<K, V> multimap = supplier.get();
106
107        if (in.peek() == JsonToken.NULL) {
108            in.nextNull();
109            return null;
110        } else if (in.peek() == JsonToken.BEGIN_ARRAY) { // see GsonBuilder.enableComplexMapKeySerialization()
111            in.beginArray(); // elements
112            while (in.hasNext()) {
113                in.beginArray(); // element
114                final K key = keyAdapter.read(in); // key
115                in.beginArray(); // values
116                while (in.hasNext()) {
117                    final V value = valueAdapter.read(in); // value
118                    multimap.put(key, value);
119                }
120                in.endArray();
121                in.endArray();
122            }
123            in.endArray();
124        } else {
125            in.beginObject();
126            while (in.hasNext()) {
127                final String keyJson = "\"" + in.nextName() + "\"";
128                final K      key     = keyAdapter.fromJson(keyJson);
129
130                in.beginArray();
131                while (in.hasNext()) {
132                    final V value = valueAdapter.read(in);
133                    multimap.put(key, value);
134                }
135                in.endArray();
136            }
137            in.endObject();
138        }
139
140        return multimap;
141    }
142
143}
